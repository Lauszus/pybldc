190 def _upload( 

191 self, 

192 binary_filename: str, 

193 timeout: float, 

194 ping_repeat: int, 

195 is_bootloader: bool = False, 

196 ) -> Generator[float, None, bool]: 

197 # 1. Try to ping the target x times to make sure we can communicate with it 

198 for i in range(ping_repeat): 

199 if self.ping(timeout=timeout): 

200 self._logger.info(f"PyBldcBase: Found VESC in {i + 1} attempt(s)") 

201 break 

202 else: 

203 self._logger.warning("Timed out waiting for ping response for VESC") 

204 return False 

205 

206 # 2. Load the binary 

207 with open(binary_filename, "rb") as f: 

208 binary_data = f.read() 

209 

210 # 3. Erase the region of memory where application will be located 

211 binary_data_len = len(binary_data) 

212 if is_bootloader: 

213 self._logger.info("PyBldcBase: Erasing bootloader") 

214 if not self._send_implementation( 214 ↛ 219line 214 didn't jump to line 219 because the condition on line 214 was never true

215 [CommPacketId.COMM_ERASE_BOOTLOADER], 

216 [], 

217 timeout=timeout, 

218 ): 

219 self._logger.error('PyBldcBase: "COMM_ERASE_BOOTLOADER" response failed for VESC') 

220 return False 

221 else: 

222 self._logger.info(f"PyBldcBase: Erasing {binary_data_len} bytes") 

223 if not self._send_implementation( 

224 [CommPacketId.COMM_ERASE_NEW_APP, *PyBldcBase._pack_uint32(binary_data_len)], 

225 [], 

226 timeout=timeout, 

227 ): 

228 self._logger.error('PyBldcBase: "COMM_ERASE_NEW_APP" response failed for VESC') 

229 return False 

230 

231 # 4. Compress the binary if needed 

232 # Note: This is not supported for bootloaders 

233 if binary_data_len > 393208: 

234 self._logger.info(f"PyBldcBase: Compression is required, as binary data is large: {binary_data_len}") 

235 binary_data = heatshrink2.encode( 

236 binary_data, 

237 window_sz2=13, 

238 lookahead_sz2=5, 

239 ) 

240 self._logger.info(f"PyBldcBase: Size after compression: {len(binary_data)}") 

241 

242 if len(binary_data) > 393208: 

243 self._logger.info("PyBldcBase: The firmware is too big even after compression") 

244 return False 

245 

246 # "0xCC" is used to indicate to the bootloader that the data is compressed 

247 # See: https://github.com/vedderb/bldc-bootloader/blob/master/main.c 

248 binary_data_len = (0xCC << 24) | len(binary_data) 

249 

250 # 5. Upload a new program chunk by chunk 

251 # For bootloaders: 

252 # Simply upload the bootloader binary 

253 # For applications: 

254 # The first 4 bytes must be set to the size of the new main program 

255 # The next 2 bytes are a CRC-checksum of the new main program 

256 # The next bytes is the program binary 

257 binary_data_crc16 = crc16_ccitt(binary_data) 

258 self._logger.info(f"PyBldcBase: Uploading binary with checksum: {binary_data_crc16}") 

259 

260 if is_bootloader: 

261 app_data = list(binary_data) 

262 else: 

263 app_data = ( 

264 PyBldcBase._pack_uint32(binary_data_len) 

265 + PyBldcBase._pack_uint16(binary_data_crc16) 

266 + list(binary_data) 

267 ) 

268 

269 chunk_size = 384 # This is the same size as the VESC Tool uses 

270 

271 if is_bootloader: 

272 # The location of the bootloader depends on the type of hardware 

273 if isinstance(self, PyBldcCan): 273 ↛ 274line 273 didn't jump to line 274 because the condition on line 273 was never true

274 if self._can_listener.hw_type == HwType.HW_TYPE_CUSTOM_MODULE: 

275 raise ValueError("HW_TYPE_CUSTOM_MODULE is not supported") 

276 elif self._can_listener.hw_type == HwType.HW_TYPE_VESC_BMS: 

277 start_addr = 0x0803E000 - 0x08020000 

278 else: 

279 start_addr = 1024 * 128 * 3 

280 else: 

281 # TODO: Determine HW type when connected via serial 

282 start_addr = 1024 * 128 * 3 

283 else: 

284 # Regular applications are always uploaded to address 0 

285 start_addr = 0 

286 

287 data_sent = 0 

288 yield 0.0 # The progress starts at 0 % 

289 for i, d in enumerate(PyBldcBase.chunks(app_data, chunk_size)): 

290 offset = start_addr + (chunk_size * i) 

291 offset_list = PyBldcBase._pack_uint32(offset) 

292 if not self._send_implementation( 

293 [CommPacketId.COMM_WRITE_NEW_APP_DATA, *offset_list, *d], 

294 expected_response=offset_list, 

295 timeout=timeout, 

296 ): 

297 self._logger.error('PyBldcBase: "COMM_WRITE_NEW_APP_DATA" response failed for VESC') 

298 return False 

299 

300 # Yield the progress in percent 

301 data_sent += len(d) 

302 yield data_sent / len(app_data) * 100.0 

303 

304 if not is_bootloader: 

305 # 6. Call the bootloader 

306 # Note: This does not have a response 

307 _ = self._send_implementation([CommPacketId.COMM_JUMP_TO_BOOTLOADER], [], timeout=0.0) 

308 

309 return True

366 def on_message_received(self, msg: can.Message) -> None: 

367 # Make sure we are not stopped and it is a valid frame 

368 if self._is_stopped or msg.is_error_frame or msg.is_remote_frame or not msg.is_extended_id: 

369 return 

370 

371 # Decode the controller ID and packet ID 

372 target_id = msg.arbitration_id & 0xFF 

373 packet_id = (msg.arbitration_id >> 8) & 0xFF 

374 

375 # We are only interested in frames to us 

376 if target_id != self._id: 

377 return 

378 

379 self._logger.debug( 

380 f"PyBldcCanListener: Received ID: 0x{msg.arbitration_id:08X}, len: {msg.dlc}, data: {list(msg.data)}" 

381 ) 

382 

383 # TODO: Add support for receiving longer packets 

384 if packet_id == CanPacketId.CAN_PACKET_PROCESS_SHORT_BUFFER: 

385 controller_id = msg.data[0] 

386 send_flag = msg.data[1] # Make sure it is a response to a packet we sent 

387 if controller_id == self._controller_id and send_flag == 1: 

388 data = list(msg.data[2:]) # Extract the data 

389 self.packet_queue.put_nowait(data) 

390 elif packet_id == CanPacketId.CAN_PACKET_PONG: 390 ↛ exitline 390 didn't return from function 'on_message_received' because the condition on line 390 was always true

391 controller_id = msg.data[0] 

392 if len(msg) > 1: 

393 self._hw_type = cast(HwType, msg.data[1]) 

394 else: 

395 # Older VESC firmwares do not report the HW type 

396 self._hw_type = HwType.HW_TYPE_VESC 

397 if controller_id == self._controller_id: 

398 self.pong_event.set()

448 def _send_implementation(self, data: list[int], expected_response: list[int], timeout: float) -> bool | None: 

449 """ 

450 See: "comm_can_send_buffer" in "bldc/comm/comm_can.c" and 

451 "packetDataToSend" in "vesc_tool/vescinterface.cpp" 

452 """ 

453 if len(data) <= 6: 

454 send_buffer: list[int] = [ 

455 self._id, 

456 0, # Process packet at receiver and send a response 

457 *data, 

458 ] 

459 

460 self._can_send_packet(CanPacketId.CAN_PACKET_PROCESS_SHORT_BUFFER, send_buffer) 

461 else: 

462 end_a = 0 

463 for i in range(0, len(data), 7): 

464 if i > 255: 

465 break 

466 end_a = i + 7 

467 

468 send_buffer = [i] 

469 send_len = 7 if i + 7 <= len(data) else len(data) - i 

470 send_buffer += data[i : i + send_len] 

471 

472 self._can_send_packet(CanPacketId.CAN_PACKET_FILL_RX_BUFFER, send_buffer) 

473 

474 # Just sleep a minimal time, so the CAN buffer does not get full 

475 # TODO: Figure out how to block until the message is sent or determine if the buffer is full 

476 time.sleep(0.0001) 

477 

478 for i in range(end_a, len(data), 6): 

479 send_buffer = PyBldcBase._pack_uint16(i) 

480 send_len = 6 if i + 6 <= len(data) else len(data) - i 

481 send_buffer += data[i : i + send_len] 

482 

483 self._can_send_packet(CanPacketId.CAN_PACKET_FILL_RX_BUFFER_LONG, send_buffer) 

484 

485 # Just sleep a minimal time, so the CAN buffer does not get full 

486 # TODO: Figure out how to block until the message is sent or determine if the buffer is full 

487 time.sleep(0.0001) 

488 

489 send_buffer = [ 

490 self._id, 

491 0, # Process packet at receiver and send a response 

492 *PyBldcBase._pack_uint16(len(data)), 

493 *PyBldcBase._pack_uint16(crc16_ccitt(data)), 

494 ] 

495 

496 self._can_send_packet(CanPacketId.CAN_PACKET_PROCESS_RX_BUFFER, send_buffer) 

497 

498 # Wait for the response 

499 return self._wait_for_packet_response( 

500 packet_queue=self._can_listener.packet_queue, 

501 comm_packet_id=cast(CommPacketId, data[0]), 

502 expected_response=expected_response, 

503 timeout=timeout, 

504 )
